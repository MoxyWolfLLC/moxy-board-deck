  async getUser(id: string): Promise<User | undefined> {
    return await db.get(`user:${id}`);
  }

  async getUserByEmail(email: string): Promise<User | undefined> {
    const userIds = await db.list("user:");
    for (const key of userIds) {
      const user = await db.get(key);
      if (user && user.email === email) return user;
    }
    return undefined;
  }

  async createUser(data: InsertUser): Promise<User> {
    const id = randomUUID();
    const user = { id, ...data, createdAt: new Date().toISOString() };
    await db.set(`user:${id}`, user);
    return user;
  }

  async updateUser(id: string, updates: Partial<User>): Promise<User | undefined> {
    const user = await this.getUser(id);
    if (!user) return undefined;
    const updated = { ...user, ...updates };
    await db.set(`user:${id}`, updated);
    return updated;
  }

  // ... implement other methods similarly
}

### Issue 3: Cookie configuration

Make sure the session cookie config in server/routes.ts has:
```typescript
cookie: {
  maxAge: 86400000 * 7,
  httpOnly: true,
  secure: false,  // Set to false for development on Replit
  sameSite: 'lax',
  path: '/'
}
```

Note: Set `secure: false` for Replit development environment since it may not be using HTTPS internally.
